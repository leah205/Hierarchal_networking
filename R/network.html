create_table <- function(df){
  ranksTableA <- df
  ranksTableA$ranks <- round(ranksTableA$ranks, 3)
  colnames(ranksTableA) <- c("Name", "Rank")
  df_t <- as.data.frame(t(ranksTableA), stringsAsFactors = F)
  df_t %>% kable(format = "html") %>% kable_styling() %>%
  column_spec(1:ncol(df_t), border_left = TRUE, border_right = TRUE) %>%
  row_spec(nrow(df_t), extra_css = "border-bottom: 1px solid black;")
}
